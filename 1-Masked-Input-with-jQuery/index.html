                              <-- script -->
                              <script type="text/javascript">
                                 
                                  $(document).ready(function () {
                                      
                                      // Apply Inputmasks here...
                                      
                                      $('#CNICNo').attr('data-inputmask', "'mask': '99999-9999999-9'").inputmask();
                                      
                                      $('#NTNNo').attr('data-inputmask', "'mask': '*******-*' , 'definitions': {'*': {'validator': '[A-Za-z0-9]'}}").inputmask();

                                      $('#GSTNo').attr('data-inputmask', "'mask': '99-9999999-9'").inputmask();

                                      $('#Phone').inputmask("(+\\9\\2) 99-9999999", {
                                          "placeholder": "_",
                                          "definitions": {
                                              'x': {
                                                  validator: "[0-9]",
                                                  cardinality: 1
                                              }
                                          }
                                      });

                                      $('#Mobile').inputmask("(+\\9\\2) 399-9999999", {
                                          "placeholder": "_",
                                          "definitions": {
                                              'x': {
                                                  validator: "[0-9]",
                                                  cardinality: 1
                                              }
                                          }
                                      });

                                      $('#Mobile2').attr('data-inputmask', "'mask': '0399-9999999'").inputmask();
                                  });
                          
                                  // Show thank you message on form submission
                                  function showThankYou() {
                                      $('form').prepend(`<div class="alert alert-success">Thank you for submitting the form!</div>`);
                                      
                                      setTimeout(function () {
                                          $('.alert').remove();
                                      }, 5000);
                                  }
                              </script>
