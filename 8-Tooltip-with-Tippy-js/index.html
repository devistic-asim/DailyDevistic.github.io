            content: attr(data-tooltip);

            top: -2.5rem;

            transform: translateY(20px);
            visibility: hidden;
        }

        .tooltip-text:hover::before,
        .tooltip-text:hover::after {
            opacity: 1;
            transform: translateY(-2px);
